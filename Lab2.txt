


#Dictionary
J = {
    # 'key' : 'value'
    'Diganta' : '22',
    'Fahim' : '23',
    'Hasin' : '24'
}
print(J)
print(J['Diganta']) # access data
print(J.get('Fahim')) # access data
print(J.keys())
print(J.values())
J.update({'Sara' : '25'}) #update - dict.update({key:value})
print(J)
J.popitem()
print(J)
J.pop('Fahim')
print(J)
del J # delete the full dictionary
